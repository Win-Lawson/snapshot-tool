function altPattern(n, size)
    turtle.up()
    move.fillRandomDown()
    skip = False
    if n == -1 then return end
    for i =1, 4 do 
        for j=1, size do
            if not skip then
                move.fillRandomDown()
            end
            skip = not skip
            turtle.forward()
        end 
    turtle.turnRight()
    end
    
    turtle.turnLeft()
    turtle.forward()
    turtle.turnLeft()
    turtle.forward()
    turtle.turnRight()
    turtle.turnRight()
    altpattern(n-1, size+2)
end
altPattern(10, 1)
